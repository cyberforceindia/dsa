1. Insertion in Array
#include<iostream>
using namespace std;
int main(){
    int arr[100],n,pos,val;
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    cin>>pos>>val;
    for(int i=n;i>=pos;i--) arr[i]=arr[i-1];
    arr[pos-1]=val;
    n++;
    for(int i=0;i<n;i++) cout<<arr[i]<<" ";
}
2. Deletion in Array
#include<iostream>
using namespace std;
int main(){
    int arr[100],n,pos;
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    cin>>pos;
    for(int i=pos-1;i<n-1;i++) arr[i]=arr[i+1];
    n--;
    for(int i=0;i<n;i++) cout<<arr[i]<<" ";
}
3. Binary Search
#include<iostream>
using namespace std;
int main(){
    int n,arr[100],key;
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    cin>>key;
    int s=0,e=n-1;
    while(s<=e){
        int mid=(s+e)/2;
        if(arr[mid]==key){
            cout<<mid;
            return 0;
        }
        else if(key<arr[mid]) e=mid-1;
        else s=mid+1;
    }
    cout<<"-1";
}
4. Linear Queue
#include<iostream>
using namespace std;
int q[100],f=-1,r=-1,n=100;
void enqueue(int x){
    if(r==n-1) return;
    if(f==-1) f=0;
    q[++r]=x;
}
void dequeue(){
    if(f==-1 || f>r) return;
    f++;
}
void display(){
    for(int i=f;i<=r;i++) cout<<q[i]<<" ";
}
int main(){
    enqueue(10); enqueue(20); enqueue(30);
    dequeue();
    display();
}
5. Circular Queue
#include<iostream>
using namespace std;
int q[5],f=-1,r=-1,n=5;
void enqueue(int x){
    if((r+1)%n==f) return;
    if(f==-1) f=0;
    r=(r+1)%n;
    q[r]=x;
}
void dequeue(){
    if(f==-1) return;
    if(f==r){ f=r=-1; return; }
    f=(f+1)%n;
}
void display(){
    int i=f;
    while(true){
        cout<<q[i]<<" ";
        if(i==r) break;
        i=(i+1)%n;
    }
}
int main(){
    enqueue(1); enqueue(2); enqueue(3);
    dequeue();
    display();
}
Sorting Techniques
6. Bubble Sort
#include<iostream>
using namespace std;
int main(){
    int n,arr[100];
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    for(int i=0;i<n-1;i++){
        for(int j=0;j<n-i-1;j++){
            if(arr[j]>arr[j+1]){
                int t=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=t;
            }
        }
    }
    for(int i=0;i<n;i++) cout<<arr[i]<<" ";
}
7. Selection Sort
#include<iostream>
using namespace std;
int main(){
    int n,arr[100];
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    for(int i=0;i<n-1;i++){
        int min=i;
        for(int j=i+1;j<n;j++){
            if(arr[j]<arr[min]) min=j;
        }
        int t=arr[i];
        arr[i]=arr[min];
        arr[min]=t;
    }
    for(int i=0;i<n;i++) cout<<arr[i]<<" ";
}
8. Insertion Sort
#include<iostream>
using namespace std;
int main(){
    int n,arr[100];
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    for(int i=1;i<n;i++){
        int key=arr[i],j=i-1;
        while(j>=0 && arr[j]>key){
            arr[j+1]=arr[j];
            j--;
        }
        arr[j+1]=key;
    }
    for(int i=0;i<n;i++) cout<<arr[i]<<" ";
}
9. Merge Sort
#include<iostream>
using namespace std;
void merge(int arr[],int l,int m,int r){
    int i=l,j=m+1,k=0,temp[100];
    while(i<=m && j<=r){
        if(arr[i]<arr[j]) temp[k++]=arr[i++];
        else temp[k++]=arr[j++];
    }
    while(i<=m) temp[k++]=arr[i++];
    while(j<=r) temp[k++]=arr[j++];
    for(int i=l,j=0;i<=r;i++,j++) arr[i]=temp[j];
}
void ms(int arr[],int l,int r){
    if(l<r){
        int m=(l+r)/2;
        ms(arr,l,m);
        ms(arr,m+1,r);
        merge(arr,l,m,r);
    }
}
int main(){
    int n,arr[100];
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    ms(arr,0,n-1);
    for(int i=0;i<n;i++) cout<<arr[i]<<" ";
}
10. Quick Sort
#include<iostream>
using namespace std;
int partition(int arr[],int l,int h){
    int pivot=arr[l],i=l+1,j=h;
    while(i<=j){
        while(arr[i]<=pivot) i++;
        while(arr[j]>pivot) j--;
        if(i<j){
            int t=arr[i]; arr[i]=arr[j]; arr[j]=t;
        }
    }
    int t=arr[l]; arr[l]=arr[j]; arr[j]=t;
    return j;
}
void qs(int arr[],int l,int h){
    if(l<h){
        int p=partition(arr,l,h);
        qs(arr,l,p-1);
        qs(arr,p+1,h);
    }
}
int main(){
    int n,arr[100];
    cin>>n;
    for(int i=0;i<n;i++) cin>>arr[i];
    qs(arr,0,n-1);
    for(int i=0;i<n;i++) cout<<arr[i]<<" ";
}
